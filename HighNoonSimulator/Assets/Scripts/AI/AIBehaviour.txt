tree("Root")
	sequence
		tree("Patrol")
		
tree("Patrol")
	fallback	
		tree("Flee")
		while not IsHealthLessThan(5)
			fallback
				tree("Attack")
				tree("Roam")
				sequence
					Stop(90.0)
					Stop(-90.0)
					Stop(145.0)
					Stop(-27.0)
				Wait(2)
tree("Attack")
	while not dead
		sequence
			seePlayer
			TargetPlayer
			LookAtTarget
			Wait(0.5)
			Fire

tree("Flee")
	sequence
		while not dead
			sequence
				IsHealthLessThan(5)
				FindHealth
				Heal

tree("Roam")
	while not dead
		while not seePlayer
			sequence
				PickDestination
				MoveToDestination
